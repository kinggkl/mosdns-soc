1. v1.0 → v2.1 变更摘要

改动v1.0v2.0v2.1原因
dial_addr 全部上游硬编码 删除 强制连接错误 IP → DoH 失效
国外主备 Google 主 → Cloudflare 备 Cloudflare 主 → Google 备 7ms vs 17ms (2.4x)
协议冗余 每上游 4 条 addr 每上游 1 DoH + 1 DoT DNS 自然解析已提供多 IP；h3 比 DoT 慢 5x 不采用
main_sequence 匹配顺序 cn → !cn → gfw gfw → !cn → cn 被墙域名优先走国外上游
规则自动更新 无 无 dcron + curl 每24h自动下载+解包+重启，SoP自包含

2.2 dns.yaml v2.0 · Cloudflare 主 · 去 dial_addr

plugins:
  # ── Cloudflare DoH + DoT（主，avg 7ms）──
  - tag: cloudflare
    type: forward
    args:
      concurrent: 3
      upstreams:
        - addr: "https://cloudflare-dns.com/dns-query"
        - addr: "tls://1dot1dot1dot1.cloudflare-dns.com"
          enable_pipeline: true

  # ── Google DoH + DoT（备，avg 17ms）──
  - tag: google
    type: forward
    args:
      concurrent: 3
      upstreams:
        - addr: "https://dns.google/dns-query"
        - addr: "tls://dns.google"
          enable_pipeline: true

  # ── 国外回退：Cloudflare → Google ──
  - tag: dns_nocn
    type: fallback
    args:
      primary: cloudflare
      secondary: google
      threshold: 500
      always_standby: true

  # ── AliDNS DoH + DoT（国内主）──
  - tag: ali
    type: forward
    args:
      concurrent: 3
      upstreams:
        - addr: "https://dns.alidns.com/dns-query"
        - addr: "tls://dns.alidns.com"
          enable_pipeline: true
          insecure_skip_verify: true

  # ── DNSPod DoH + DoT（国内备）──
  - tag: dnspod
    type: forward
    args:
      concurrent: 3
      upstreams:
        - addr: "https://doh.pub/dns-query"
        - addr: "tls://dot.pub"
          enable_pipeline: true

  # ── 国内回退：AliDNS → DNSPod ──
  - tag: dns_cn
    type: fallback
    args:
      primary: ali
      secondary: dnspod
      threshold: 500
      always_standby: true

  # ── ECS 处理 ──
  - tag: no_ecs
    type: ecs_handler
    args:
      forward: false
      preset: ""
      send: false
      mask4: 24
      mask6: 48

  - tag: ecs_cn
    type: ecs_handler
    args:
      forward: false
      preset: "202.120.2.100"
      send: false
      mask4: 24
      mask6: 48

  # ── Reject 响应 ──
  - tag: reject_3
    type: sequence
    args: [{ exec: reject 3 }]

  - tag: reject_2
    type: sequence
    args: [{ exec: reject 2 }]

  - tag: reject_5
    type: sequence
    args: [{ exec: reject 5 }]

2.3 dat_exec.yaml 分流核心

plugins:
  # ── 数据源 ──
  - tag: geoip_private           type: ip_set      files: ["/etc/mosdns/dat/geoip_private.txt"]
  - tag: geoip_cn                type: ip_set      files: ["/etc/mosdns/dat/geoip_cn.txt"]
  - tag: geosite_cn              type: domain_set  files: ["/etc/mosdns/dat/geosite_cn.txt"]
  - tag: geosite_ads_all         type: domain_set  files: ["/etc/mosdns/dat/geosite_category-ads-all.txt"]
  - tag: geosite_gfw             type: domain_set  files: ["/etc/mosdns/dat/geosite_gfw.txt"]
  - tag: geosite_location_not_cn type: domain_set  files: ["/etc/mosdns/dat/geosite_geolocation-!cn.txt"]

  # ── 双缓存 ──
  - tag: cache_lan   type: cache   size: 8192   lazy_cache_ttl: 86400
  - tag: cache_wan   type: cache   size: 131072  lazy_cache_ttl: 86400  dump_file: ./cache.dump  dump_interval: 120

  # ── TTL ──
  - tag: ttl_1h   type: sequence  args: [{ exec: ttl 3600 }]
  - tag: ttl_5m   type: sequence  args: [{ exec: ttl 300 }]

  # ── 结果处理 ──
  - tag: pre_handle     type: sequence  args: [{ exec: $ttl_1h }, { exec: accept }]
  - tag: main_handle    type: sequence  args: [{ exec: $ttl_5m }, { exec: accept }]
  - tag: has_resp_pre   type: sequence  args: [{ matches: has_resp, exec: goto pre_handle }]
  - tag: has_resp_main  type: sequence  args: [{ matches: has_resp, exec: goto main_handle }]

  # ── 查询路由 ──
  - tag: query_cn
    type: sequence
    args:
      - exec: $ecs_cn
      - exec: $dns_cn
      - matches: "!resp_ip $geoip_cn"     exec: drop_resp

  - tag: query_nocn
    type: sequence
    args:
      - exec: $dns_nocn
      - matches: "resp_ip $geoip_cn"       exec: drop_resp

  - tag: query_lan
    type: sequence
    args:
      - exec: $cache_lan
      - matches: has_resp     exec: return
      - exec: $no_ecs
      - exec: $dns_cn

  - tag: query_other
    type: sequence
    args:
      - exec: $no_ecs
      - exec: $dns_nocn

  # ── 拒绝 ──
  - tag: reject_qtype65      type: sequence  args: [{ exec: query_summary reject_qtype65 }, { exec: $reject_3 }]
  - tag: reject_null_domain  type: sequence  args: [{ exec: query_summary reject_null_domain }, { exec: $reject_3 }]
  - tag: reject_ad           type: sequence  args: [{ exec: query_summary reject_ad }, { exec: $reject_3 }]

  # ── 预处理 ──
  - tag: pre_sequence
    type: sequence
    args:
      - matches: "qtype 65"          exec: $reject_qtype65
      - matches: "qname keyword::"   exec: $reject_null_domain
      - matches: "qtype 12"          exec: $query_other
      - matches: "qtype 255"         exec: $query_other
      - matches: "ptr_ip $geoip_private"  exec: $query_lan
      - exec: jump has_resp_pre

  # ── 主流程 ──
  - tag: main_sequence
    type: sequence
    args:
      - matches: "qname $geosite_ads_all"         exec: $reject_ad
      - exec: $cache_wan
      - exec: jump has_resp_main
      - matches: "qname $geosite_gfw"             exec: $query_nocn
      - exec: jump has_resp_main
      - matches: "qname $geosite_location_not_cn" exec: $query_nocn
      - exec: jump has_resp_main
      - matches: "qname $geosite_cn"              exec: $query_cn
      - exec: jump has_resp_main
      - exec: $no_ecs
      - exec: $query_nocn
      - exec: jump has_resp_main

  # ── 顶层 ──
  - tag: sequence
    type: sequence
    args:
      - exec: $pre_sequence
      - exec: $main_sequence

2.4 docker-compose.yml

services:
  mosdns:
    build: { context: ., dockerfile: Dockerfile }
    image: mosdns:v5.3.4
    container_name: mosdns
    restart: always
    network_mode: host
    volumes:
      - ./config.yaml:/etc/mosdns/config.yaml:ro
      - ./dat_exec.yaml:/etc/mosdns/dat_exec.yaml:ro
      - ./dns.yaml:/etc/mosdns/dns.yaml:ro
      - ./dat:/etc/mosdns/dat:ro
      - ./cache:/etc/mosdns/cache
    logging:
      driver: json-file
      options: { max-size: "50m", max-file: "3" }
